if statement 
<?php
$age = 20;

if ($age >= 18) {
    echo "You are an adult";
}
?>

if else statement 
<?php
$temperature = 25;

if ($temperature > 30) {
    echo "It is hot";
} else {
    echo "It is not too hot";
}
?>

if...elseif...else statement 
<?php
$score = 89;

if ($score >= 90) {
    echo "Grade: A\n";
} elseif ($score >= 80) {
    echo "Grade: B";
} elseif ($score >= 70) {
    echo "Grade: C";
} else {
    echo "Grade: F";
}
?>
